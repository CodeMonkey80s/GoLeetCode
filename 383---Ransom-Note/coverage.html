package solution383

import "fmt"

// ============================================================================
// 383. Ransom Note
// URL: https://leetcode.com/problems/ransom-note/
// ============================================================================

func canConstruct(ransomNote string, magazine string) bool {
        // count1 := len(ransomNote)
        // count2 := len(magazine)
        m := make(map[byte]int)
        for _, v := range ransomNote {
                ch := byte(v)
                _, ok := m[ch]
                if !ok {
                        m[ch] = 1
                } else {
                        m[ch]++
                }
        }
        fmt.Printf("MAP (1): %v\n", m)
        for _, v := range magazine {
                ch := byte(v)
                _, ok := m[ch]
                if ok {
                        fmt.Printf("found char %c in map: %v\n", ch, m)
                        m[ch]--
                        if m[ch] < 0 {
                                fmt.Printf("less than 0!\n")
                                delete(m, ch)
                        }
                }
        }
        fmt.Printf("MAP (2): %v\n", m)
        return len(m) == 0
}
